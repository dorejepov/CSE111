-- Broker Table
CREATE TABLE Brokers (
    broker_id INTEGER PRIMARY KEY AUTOINCREMENT,
    company_name TEXT NOT NULL,
    mc_number TEXT NOT NULL,
    dot_number TEXT NOT NULL,
    credit_score NUMERIC(3,1),
    load_number INTEGER
);

-- Truck Table
CREATE TABLE Trucks (
    truck_id INTEGER PRIMARY KEY AUTOINCREMENT,
    make TEXT NOT NULL,
    model TEXT NOT NULL,
    vin TEXT UNIQUE NOT NULL,
    license_plate TEXT NOT NULL,
    registration TEXT,
    physical_insurance TEXT,
    owner TEXT,
    expenses NUMERIC(10,2),
    driver_id INTEGER UNIQUE,  -- Foreign key for one-to-one relationship
    FOREIGN KEY (driver_id) REFERENCES Drivers(driver_id)
);

-- Carrier Table
CREATE TABLE Carriers (
    carrier_id INTEGER PRIMARY KEY AUTOINCREMENT,
    carrier_name TEXT NOT NULL,
    insurance_policy_number TEXT,
    dot_number TEXT NOT NULL,
    mc_number TEXT NOT NULL,
    number_of_trucks INTEGER
);

-- Insurance Table
CREATE TABLE Insurance (
    insurance_id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL,
    phone_number TEXT,
    expiry_date DATE,
    insurance_agent TEXT,
    agent_phone_number TEXT
);

-- Dispatcher Table
CREATE TABLE Dispatchers (
    dispatcher_id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL,
    phone_number TEXT,
    carrier_id INTEGER,
    payment_type TEXT,
    bank_info TEXT,
    FOREIGN KEY (carrier_id) REFERENCES Carriers(carrier_id)
);

-- Driver Table
CREATE TABLE Drivers (
    driver_id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL,
    phone_number TEXT,
    carrier_id INTEGER,
    payment_type TEXT,
    bank_info TEXT,
    medical_card TEXT,
    FOREIGN KEY (carrier_id) REFERENCES Carriers(carrier_id)
);

-- Many-to-Many Relationship Tables
-- Broker_Dispatcher Table
CREATE TABLE Broker_Dispatcher (
    broker_dispatcher_id INTEGER PRIMARY KEY AUTOINCREMENT,
    broker_id INTEGER,
    dispatcher_id INTEGER,
    FOREIGN KEY (broker_id) REFERENCES Brokers(broker_id),
    FOREIGN KEY (dispatcher_id) REFERENCES Dispatchers(dispatcher_id)
);

-- Broker_Driver Table
CREATE TABLE Broker_Driver (
    broker_driver_id INTEGER PRIMARY KEY AUTOINCREMENT,
    broker_id INTEGER,
    driver_id INTEGER,
    FOREIGN KEY (broker_id) REFERENCES Brokers(broker_id),
    FOREIGN KEY (driver_id) REFERENCES Drivers(driver_id)
);

-- Broker_Carrier Table
CREATE TABLE Broker_Carrier (
    broker_carrier_id INTEGER PRIMARY KEY AUTOINCREMENT,
    broker_id INTEGER,
    carrier_id INTEGER,
    FOREIGN KEY (broker_id) REFERENCES Brokers(broker_id),
    FOREIGN KEY (carrier_id) REFERENCES Carriers(carrier_id)
);

-- Dispatcher_Driver Table
CREATE TABLE Dispatcher_Driver (
    dispatcher_driver_id INTEGER PRIMARY KEY AUTOINCREMENT,
    dispatcher_id INTEGER,
    driver_id INTEGER,
    FOREIGN KEY (dispatcher_id) REFERENCES Dispatchers(dispatcher_id),
    FOREIGN KEY (driver_id) REFERENCES Drivers(driver_id)
);
